set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -Wall -Wextra")

project(MusicVi)

add_executable(${PROJECT_NAME} main.c)

target_link_libraries(${PROJECT_NAME} raylib winmm gdi32)
